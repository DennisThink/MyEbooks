25  public:

28  // The PosixLogger instance takes ownership of the file handle.

31  }

33  ~PosixLogger() override {

37  void Logv(const char* format, va_list arguments) override {
38  // Record the time as close to the Logv() call as possible.
39  struct ::timeval now_timeval;
40  ::gettimeofday(&now_timeval, nullptr);
41  const std::time_t now_seconds = now_timeval.tv_sec;
42  struct std::tm now_components;
43  ::localtime_r(&now_seconds, &now_components);
44 
45  // Record the thread ID.
46  constexpr const int kMaxThreadIdSize = 32;
47  std::ostringstream thread_stream;
48  thread_stream << std::this_thread::get_id();
49  std::string thread_id = thread_stream.str();
50  if (thread_id.size() > kMaxThreadIdSize) {
51  thread_id.resize(kMaxThreadIdSize);
52  }
53 
54  // We first attempt to print into a stack-allocated buffer. If this attempt
55  // fails, we make a second attempt with a dynamically allocated buffer.
56  constexpr const int kStackBufferSize = 512;
57  char stack_buffer[kStackBufferSize];
58  static_assert(sizeof(stack_buffer) == static_cast<size_t>(kStackBufferSize),
59  "sizeof(char) is expected to be 1 in C++");
60 
61  int dynamic_buffer_size = 0; // Computed in the first iteration.
62  for (int iteration = 0; iteration < 2; ++iteration) {
63  const int buffer_size =
64  (iteration == 0) ? kStackBufferSize : dynamic_buffer_size;
65  char* const buffer =
66  (iteration == 0) ? stack_buffer : new char[dynamic_buffer_size];
67 
68  // Print the header into the buffer.
69  int buffer_offset = snprintf(
70  buffer, buffer_size,
71  "%04d/%02d/%02d-%02d:%02d:%02d.%06d %s",
72  now_components.tm_year + 1900,
73  now_components.tm_mon + 1,
74  now_components.tm_mday,
75  now_components.tm_hour,
76  now_components.tm_min,
77  now_components.tm_sec,
78  static_cast<int>(now_timeval.tv_usec),
79  thread_id.c_str());
80 
81  // The header can be at most 28 characters (10 date + 15 time +
82  // 3 spacing) plus the thread ID, which should fit comfortably into the
83  // static buffer.
84  assert(buffer_offset <= 28 + kMaxThreadIdSize);
85  static_assert(28 + kMaxThreadIdSize < kStackBufferSize,
86  "stack-allocated buffer may not fit the message header");
87  assert(buffer_offset < buffer_size);
88 
89  // Print the message into the buffer.
90  std::va_list arguments_copy;
91  va_copy(arguments_copy, arguments);
92  buffer_offset += std::vsnprintf(buffer + buffer_offset,
93  buffer_size - buffer_offset, format,
94  arguments_copy);
95  va_end(arguments_copy);
96 
97  // The code below may append a newline at the end of the buffer, which
98  // requires an extra character.
99  if (buffer_offset >= buffer_size - 1) {
100  // The message did not fit into the buffer.
101  if (iteration == 0) {
102  // Re-run the loop and use a dynamically-allocated buffer. The buffer
103  // will be large enough for the log message, an extra newline and a
104  // null terminator.
105  dynamic_buffer_size = buffer_offset + 2;
106  continue;
107  }
108 
109  // The dynamically-allocated buffer was incorrectly sized. This should
110  // not happen, assuming a correct implementation of (v)snprintf. Fail
111  // in tests, recover by truncating the log message in production.
112  assert(false);
113  buffer_offset = buffer_size - 1;
114  }
115 
116  // Add a newline if necessary.
117  if (buffer[buffer_offset - 1] != '\n') {
118  buffer[buffer_offset] = '\n';
119  ++buffer_offset;
120  }
121 
122  assert(buffer_offset <= buffer_size);
123  std::fwrite(buffer, 1, buffer_offset, fp_);
124  std::fflush(fp_);
125 
126  if (iteration != 0) {
127  delete[] buffer;
128  }
129  break;
130  }
131  }
132 
133  private:
134  std::FILE* const fp_;
135 };
